#include<stdio.h>
void sortarray(int arr[],int size)
{
    int low=0,high=size-1,mid=0;
    while(mid<=high)
    {
        switch(arr[mid])
        {
            case 0:
            {
                swap(&arr[low],&arr[mid]);
                low++;
                mid++;
                 break;
            }
           case 1:
           {
               mid++;
               break;
           }
           case 2:
           {
               swap(&arr[mid],&arr[high]);
               high--;
               break;
           }
        }
    }
    
}
void swap(int *a,int *b)
{
    int temp=*a;
    *a=*b;
    *b=temp;
}
int main()
{
    int arr[]={1,2,0,1,2,0,1,0,1,2}; int i;
    int size=sizeof(arr)/sizeof(arr[0]);
    sortarray(arr,size);
    
    printf("after sorting elemets are\n");
    for(i=0;i<size;i++)
    {
        printf("%d\n",arr[i]);
    }
}
