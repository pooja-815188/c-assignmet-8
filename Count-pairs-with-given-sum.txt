#include<stdio.h>
void pairs(int arr[],int size,int k)
{
    int i,j,count=1;
    for(i=0;i<size;i++)
    {
        for(j=i+1;j<size;j++)
        if(arr[i]+arr[j]==k)
        {
            count++;
        }
    }
    printf("pairs of elemt=%d\n",count);
}
int main()
{
    int arr[]={10,2,30,4,5,6,1};
    int size=sizeof(arr)/sizeof(arr[0]);
    int k=10;
    pairs(arr,size,k);
}
